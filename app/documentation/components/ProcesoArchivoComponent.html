import { Component, ViewChild, OnInit, TemplateRef, ElementRef } from '@angular/core';
import swal from 'sweetalert2';

import { Observable } from 'rxjs/Observable';
import { zip } from 'rxjs/observable/zip';

import { MensajeService } from '../../../app-core/services/app-admin/mensaje/mensaje.service';
import { InscrAspiranteService } from '../../../app-core/services/inscr-aspirante/inscr-aspirante.service';


@Component({
  selector: 'app-proceso-archivo',
  templateUrl: './proceso-archivo.component.html',
  styleUrls: ['./proceso-archivo.component.scss']
})
export class ProcesoArchivoComponent {

  files: File;
  lstFaltantes: Array<any>;
  lstFaltantesColumn = [
    { prop: 'codigo', name: 'CODIGO', minWidth: 130 },
    { prop: 'tipo_documento_saber', name: 'TIPO DOC', minWidth: 50 },
    { prop: 'documento_icfes', name: 'NUM DOC', minWidth: 130 },
    { prop: 'nom_ape', name: 'NOMBRES Y APELLIDOS', minWidth: 300 }
  ];


  constructor(
    public serviceMensaje: MensajeService,
    private serviceInscrAspirante: InscrAspiranteService,
  ) {
    this.lstFaltantes = [];
  }

  /**
   * 
   */
  ngOnInit() {

  }


  /**
   * 
   */
  limpiarVariables() {
    this.lstFaltantes = [];
  }


  /**
   * 
   */
  generarPlano() {
    this.serviceMensaje.procesando();
    this.limpiarVariables();

    this.serviceInscrAspirante.getAspirantes()
      .subscribe(
        (data: any) => {
          try {
            this.serviceMensaje.cerrarMensaje();

            let nomArchivo = data.headers.get('x-filename') != null ? data.headers.get('x-filename') : 'archivo.txt';

            var blob = new Blob([data.body], { type: "text/plain" });
            if (window.navigator && window.navigator.msSaveOrOpenBlob) {
              window.navigator.msSaveOrOpenBlob(blob, nomArchivo);
            } else {
              var a = document.createElement('a');
              a.href = URL.createObjectURL(blob);
              a.download = nomArchivo;
              document.body.appendChild(a);
              a.click();
              document.body.removeChild(a);
            }
          } catch (error) {
            this.serviceMensaje.enviarMensaje('Error', error, 'e');
          }

        },
        error => {
          this.serviceMensaje.mensajeErrorServidor(error);
        }
      );
  }


  /**
   * 
   * @param valido 
   */
  uploadFiles(valido) {
    this.serviceMensaje.procesando();
    this.limpiarVariables();

    this.serviceInscrAspirante.enviarArchivoCarga(this.files)
      .subscribe(
        (data: any) => {
          this.serviceMensaje.cerrarMensaje();
          console.log('Resultado :::> ', data);
          if (data.faltantes == true) {
            this.lstFaltantes = data.lstFaltantes;
            this.serviceMensaje.mensajeServidor(data, 'w');
          }
        },
        error => {
          this.serviceMensaje.mensajeErrorServidor(error);
        }
      );
  }


}
