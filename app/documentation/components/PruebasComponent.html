import { Component, ViewChild, OnInit, TemplateRef } from '@angular/core';
import { MensajeService } from '../../../app-core/services/app-admin/mensaje/mensaje.service';
import { PruebaAdmisionService } from '../../../app-core/services/prueba-admision/prueba-admision.service';
import { HelpersService } from '../../../app-core/services/app-admin/helpers/helpers.service';



@Component({
  selector: 'app-pruebas',
  templateUrl: './pruebas.component.html',
  styleUrls: ['./pruebas.component.scss']
})
export class PruebasComponent {

  lstPruebas: Array<any>;
  lstAspirantes: Array<any>;
  pruebaEscojida: any;
  esCalificar: boolean;
  lstAreas: Array<any>;
  lstAreasCalificados: Array<any>;
  idPrueba: Number;



  constructor(
    public serviceMensaje: MensajeService,
    private servicePruebaAdmision: PruebaAdmisionService,
    private serviceHelpers: HelpersService,
  ) {
    this.lstPruebas = [];
    this.lstAspirantes = [];
    this.esCalificar = false;
    this.lstAreas = [];
    this.lstAreasCalificados=[];
    this.idPrueba = 0;
    this.pruebaEscojida = null;

  }


  ngOnInit() {
    this.obtenerPruebas();
  }


  limpiarVariables() {
    this.lstPruebas = [];
    this.lstAspirantes = [];
    this.esCalificar = false;
    this.lstAreas = [];
    this.idPrueba = 0;
    this.pruebaEscojida = null;
    this.lstAreasCalificados=[];
  }


  obtenerPruebas() {
    this.serviceMensaje.procesando();
    this.limpiarVariables();

    this.servicePruebaAdmision.obtener_listado()
      .subscribe(
        (pruebas: any) => {
          try {
            if (pruebas.length > 0) {
              console.log('toca escojer');
              this.lstPruebas = pruebas;

            } else {
              throw new Error('Ud no tiene asignado una prueba de admision');
            }
            this.serviceMensaje.cerrarMensaje();
          } catch (error) {
            if (error.message) {
              this.serviceMensaje.enviarMensaje('Error', error.message, 'e');
            } else {
              this.serviceMensaje.enviarMensaje('Error', 'Error al obtener el listado de aspirantes, intentelo nuevamente por favor', 'e');
            }
          }

        },
        error => {
          this.serviceMensaje.mensajeErrorServidor(error);
        }
      );
  }

  seleccionarPrueba() {
    this.pruebaEscojida = this.serviceHelpers.buscarObjetoByKey(this.lstPruebas, 'id_prueba', this.idPrueba);
    if (this.pruebaEscojida) {
        console.log(this.pruebaEscojida);
        this.pruebaEscojida.fecha_formateada = this.serviceHelpers.formatearFecha(this.pruebaEscojida.fecha_prueba, 'S');
        if (this.pruebaEscojida.estado_prueba === 'C') {
          this.pruebaEscojida.fecha_formateada_calif = this.serviceHelpers
          .formatearFecha(this.pruebaEscojida.fecha_registro_calificacion, 'S');
          this.serviceMensaje.procesando();


          var listadoAspirantes= new Array();
          this.pruebaEscojida.listado.forEach(item=>{
              listadoAspirantes.push(item.id_aspirante_oferta_programa);
          });


          this.servicePruebaAdmision.obtenerListaCalificados(listadoAspirantes)
            .subscribe(
              (listado: any)=>{
                try{

                  console.log(listado);

                  this.pruebaEscojida.listado.forEach(item => {
                    listado.forEach(iobtenido => {
                      if(item.id_aspirante_oferta_programa== iobtenido.id_aspirante_oferta_programa){
                        item.lstAreasCalificados= iobtenido.lstAreas;
                      }
                    });
                  });

                  this.servicePruebaAdmision.obtenerAreas(this.pruebaEscojida)
                    .subscribe(
                      (areas:any)=>{

                        try{


                          this.lstAreasCalificados = areas;
                          this.serviceMensaje.cerrarMensaje();

                        }
                        catch (error){
                          if (error.message) {
                            this.serviceMensaje.enviarMensaje('Error', error.message, 'e');
                          } else {
                            this.serviceMensaje.enviarMensaje('Error', 'Error al obtener el listado de aspirantes, intentelo nuevamente por favor', 'e');
                          }
                        }
                      },
                      error=>{
                        this.serviceMensaje.mensajeErrorServidor(error);
                      }
                    )
                 
                }
                catch(error){
                  this.serviceMensaje.enviarMensaje('Error',
                  'Error al obtener el listado de calificaciones', 'e');
                }
              },
              error =>{
                this.serviceMensaje.mensajeErrorServidor(error);
              }
            )
        }
    } else {
      this.serviceMensaje.enviarMensaje('Error', 'Seleccione una prueba valida', 'e');
    }
  }

  calificar() {
    this.serviceMensaje.procesando();
    if (this.pruebaEscojida) {
      this.servicePruebaAdmision.obtenerAreas(this.pruebaEscojida)
        .subscribe(
          (areas: any) => {
            try {
              this.esCalificar = true;
              this.lstAreas = areas;

              const listacalificaciones = new Array();

              this.lstAreas.forEach(item => {
                const auxiliar = {
                  id_detalle_prueba_admision: item.id_detalle_prueba_admision,
                  porcentaje: item.porcentaje,
                  valor: ''
                };
                listacalificaciones.push(auxiliar);
              });

              this.pruebaEscojida.listado.forEach(element => {
                element.lstAreas = JSON.parse(JSON.stringify(listacalificaciones));
              });



              this.serviceMensaje.cerrarMensaje();
            } catch (error) {
              this.serviceMensaje.enviarMensaje('Error',
              'Error al obtener el listado de areas de la prueba, intentelo nuevamente por favor', 'e');
            }
          },
          error => {
            this.serviceMensaje.mensajeErrorServidor(error);
          });

    } else {
      this.serviceMensaje.enviarMensaje('Error', 'Error al consultar la información, por favor actualice la pagina', 'e');
    }

  }

  cancelar() {
    this.esCalificar = false;
    this.lstAreas = [];
    this.lstAreasCalificados=[];
    this.pruebaEscojida.listado.forEach(element => {
      element.lstAreas = [];
    });
  }

  guardar(esValido) {
    if (esValido) {
      this.serviceMensaje.procesando();
      if (this.esCalificar) {
        if (this.pruebaEscojida.listado) {

          this.servicePruebaAdmision.guardarCalificaciones(this.pruebaEscojida)
          .subscribe(
            (resultado: any ) => {
              try {
                  this.serviceMensaje.cerrarMensaje();
                  this.serviceMensaje.enviarMensajeFuncion("Registros actualizados","Los registros de las pruebas de admisión fueron actualizados en el sistema","s",function(){
                    window.location.reload();
                  });
                  
              } catch (error) {
                if (error.message) {
                  this.serviceMensaje.enviarMensaje('Error', error.message, 'e');
                } else {
                  this.serviceMensaje.enviarMensaje('Error',
                  'Error al obtener el listado de areas de la prueba, intentelo nuevamente por favor', 'e');
                }
              }

            },
            error => {
              this.serviceMensaje.mensajeErrorServidor(error);
            }
          );

        } else {
          this.serviceMensaje.enviarMensaje('Error', 'Error al actualizar los registros', 'e');
        }
      } else {
        this.serviceMensaje.enviarMensaje('Error', 'Error al actualizar los registros', 'e');
      }
    } else {
      this.serviceMensaje.enviarMensaje('Error', 'No ha diligenciado todas las calificaciones, revise los datos', 'w');
    }

  }

  regresar() {
    this.lstAspirantes = [];
    this.esCalificar = false;
    this.lstAreas = [];
    this.pruebaEscojida = null;
    this.idPrueba = 0;
    this.lstAreasCalificados=[];
  }
}
